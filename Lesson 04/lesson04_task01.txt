print("Введите через пробел длины сторон треугольника. После завершения ввода нажмите клавишу Enter")
a, b, c = map(float, input().split())
p = a + b + c                                           # вычисляем периметр
p2 = p / 2                                              # считаем полупериметр
s = (p2 * (p2 - a) * (p2 - b) * (p2 - c)) ** 0.5        # вычисляем площадь по формуле Герона
print("Периметр треугольника =", p)
print("Площадь треугольника по формуле Герона =", s)
